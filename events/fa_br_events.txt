				name = ayachuco_treaty_bol_medium_money

				is_decaying = no
			}

			add_modifier = {

			s:STATE_AMAZONAS.region_state:BOL = {

			}
		}

			country = c:BRZ
			value = 10
		}
		ai_chance = {
			base = 0
			modifier = {
				add = 50
				c:BRZ = {
					var:brz_bolivia_money = 3
				}

					var:brz_bolivia_money = 2

					c:BRZ = {
						var:brz_bolivia_money = 3

				add = 50
				in_default = yes
			}

				add = 25
				taking_loans = yes
			}
		}
	}
