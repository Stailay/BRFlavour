je_br_claim_uruguay = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	is_shown_when_inactive = {
	}

	possible = {
	}

	immediate = {
		set_variable = {
			name = uru_claim_var
			value = 0
		}
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					ig:ig_armed_forces = {
						ig_clout >= 0.12
						ig_approval >= 5
						is_in_government = yes
					}
					has_insurrectionary_interest_groups = no
					is_at_war = no
					owns_entire_state_region = STATE_RIO_DE_JANEIRO
					owns_entire_state_region = STATE_RIO_GRANDE_DO_SUL
					OR = {
						c:USA = {
							relations:root >= relations_threshold:amicable
						}
						c:FRA = {
							relations:root >= relations_threshold:amicable
						}
						c:GBR = {
							relations:root >= relations_threshold:amicable
						}
					}
				}
				change_variable = {
					name = uru_claim_var
					add = 1
				}
			}
		}
	}

	complete = {
		# If-else clause to avoid errors because of missing variable before JE activates
		trigger_if = {
			limit = { exists = var:uru_claim_var }
			scope:journal_entry = { is_goal_complete = yes }
		}
		trigger_else = {
			custom_tooltip = je_br_claim_uruguay_goal
		}
		ig:ig_armed_forces = {
			ig_clout >= 0.15
			ig_approval >= 5
			is_in_government = yes
		}
		has_insurrectionary_interest_groups = no
		is_at_war = no
		owns_entire_state_region = STATE_RIO_DE_JANEIRO
		owns_entire_state_region = STATE_RIO_GRANDE_DO_SUL
		OR = {
			c:USA = {
				relations:root >= relations_threshold:amicable
			}		
			c:FRA = {
				relations:root >= relations_threshold:amicable
			}
			c:GBR = {
				relations:root >= relations_threshold:amicable
			}
		}
	}

	on_complete = {
		set_variable = uruguay_claim_complete
		trigger_event = {
			id = brclaims.1
		}
	}

	current_value = {
		value = root.var:uru_claim_var
	}

	goal_add_value = {
		value = 12
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
